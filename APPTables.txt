CREATE TABLE Users (id integer PRIMARY KEY auto_increment, uName varchar(20) UNIQUE not null);

CREATE TABLE Communities (id integer PRIMARY KEY auto_increment, cName varchar(20) UNIQUE not null);

CREATE TABLE Roles (id integer PRIMARY KEY auto_increment, rName varchar(20) UNIQUE not null, aLevel integer not null);

CREATE TABLE Events (id integer PRIMARY KEY auto_increment, eName varchar(30) not null, start DATE not null, end DATE not null, cID integer not null);
